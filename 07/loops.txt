17. Write logic for **rate-limiting**:

    * Allow a maximum of 5 requests
    * Block further processing once limit is exceeded
let requestCount = 0;
const MAX_REQUESTS = 5;

while (true) {
  if (requestCount >= MAX_REQUESTS) {
    console.log("Rate limit exceeded");
    break;
  }

  prompt("Enter request");
  requestCount++;
}


18. Write a loop that:

    * Processes a list of tasks
    * Skips completed tasks
    * Stops if a critical task fails
let tasks = [
  { name: "Login", completed: true,  critical: false, status: "success" },
  { name: "Fetch Data", completed: false, critical: false, status: "success" },
  { name: "Payment", completed: false, critical: true,  status: "failed" },
  { name: "Send Email", completed: false, critical: false, status: "success" }
];

for (let i = 0; i < tasks.length; i++) {
  // Skip completed tasks
  if (tasks[i].completed) continue;

  // Stop if a critical task fails
  if (tasks[i].critical && tasks[i].status === "failed") {
    console.log("Critical task failed. Stopping process.");
    break;
  }

  // Process task
  console.log("Processing task:", tasks[i].name);
}
